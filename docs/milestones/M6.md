## 设计提示

```gdscript
extends Camera3D

@export var target_path: NodePath
@export var min_distance := 80.0
@export var max_distance := 500.0
@export var rotate_speed := 0.5
@export var zoom_speed := 5.0

var _target: Node3D
var _yaw := 0.0
var _pitch := 0.0
var _distance := 150.0

func _ready():
    _target = get_node(target_path)
    _update_transform()

func _unhandled_input(event):
    if event is InputEventMouseMotion and event.button_mask & MOUSE_BUTTON_MASK_LEFT:
        _yaw -= event.relative.x * rotate_speed * 0.01
        _pitch = clamp(_pitch - event.relative.y * rotate_speed * 0.01, -89, 89)
        _update_transform()
    if event is InputEventMouseButton:
        if event.button_index == MOUSE_BUTTON_WHEEL_UP:
            _distance = max(min_distance, _distance - zoom_speed)
            _update_transform()
        elif event.button_index == MOUSE_BUTTON_WHEEL_DOWN:
            _distance = min(max_distance, _distance + zoom_speed)
            _update_transform()

func _update_transform():
    # 球坐标 → 位置
    var rad_yaw = deg_to_rad(_yaw)
    var rad_pitch = deg_to_rad(_pitch)
    var x = _distance * cos(rad_pitch) * sin(rad_yaw)
    var y = _distance * sin(rad_pitch)
    var z = _distance * cos(rad_pitch) * cos(rad_yaw)
    global_position = _target.global_position + Vector3(x, y, z)
    look_at(_target.global_position)
```

## 验收
- 鼠标拖动时地球在视野里旋转
- 滚轮缩放（不会穿透地球或飞到外太空）
